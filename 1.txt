const GEMINI_MODEL = "gemini-flash-lite-latest";

function askAI(prompt) {

  const apiKey = PropertiesService
    .getScriptProperties()
    .getProperty("GEMINI_API_KEY");

  if (!apiKey) {
    throw new Error(
      "GEMINI_API_KEY script property is missing."
    );
  }

  const url =
    "https://generativelanguage.googleapis.com/v1beta/models/" +
    GEMINI_MODEL +
    ":generateContent";

  const payload = {

    contents: [
      {
        role: "user",
        parts: [
          {
            text: prompt
          }
        ]
      }
    ],

    generationConfig: {
      temperature: 0.4,
      topP: 0.95,
      topK: 40,
      maxOutputTokens: 1024
    }
  };

  let response;

  try {

    response = UrlFetchApp.fetch(url, {

      method: "post",

      contentType: "application/json",

      muteHttpExceptions: true,

      headers: {
        "x-goog-api-key": apiKey
      },

      payload: JSON.stringify(payload)

    });

  } catch (e) {

    throw new Error(
      "Could not reach Gemini: " + e.message
    );

  }

  const status = response.getResponseCode();
  const body = response.getContentText();

  let json;

  try {

    json = JSON.parse(body);

  } catch (e) {

    throw new Error(
      "Gemini returned invalid JSON.\n\n" + body
    );

  }

  if (status !== 200) {

    const message =
      json.error &&
      json.error.message
        ? json.error.message
        : body;

    throw new Error(message);

  }

  if (
    !json.candidates ||
    !json.candidates.length
  ) {

    throw new Error(
      "Gemini returned no candidates."
    );

  }

  const text =
    json.candidates[0]
      .content
      .parts[0]
      .text;

  if (!text) {

    throw new Error(
      "Gemini returned an empty response."
    );

  }

  return text.trim();

}

 * ------------------------------------------------------------
 * EXTENDED — new fields added for Email.gs / AI.gs.
 * Nothing below the "EXISTING CODE" marker was changed. Everything
 * under "NEW — added for Communication Center" is additive:
 * new helper functions + new keys appended to the object returned
 * by mapApplication(). No existing field was removed or renamed.
 *
 * NOTE: Database.gs (getColumnMap / getHeaders / getRows) was left
 * untouched, as instructed. The new fields below are resolved with
 * their own header lookup (getExtendedColumnMap_) so nothing in
 * Database.gs needs to change.
 * ------------------------------------------------------------
 */

// ============================================================

function mapApplication(row, id) {

  const col = getColumnMap();

  const isSTUApplication =
    String(row[col.university]).trim() ===
    "Slovak University of Technology in Bratislava";

  let university = row[col.university];
  let faculty = row[col.faculty];
  let mobility = row[col.mobility];
  let studyLevel = row[col.studyLevel];
  let position = row[col.position];

  if (isSTUApplication) {

    university = row[col.foreignUniversity];

    faculty = row[col.foreignFaculty];

    mobility = row[col.foreignMobility];

    studyLevel = row[col.foreignStudyLevel];

    position = row[col.foreignPosition];

  }

  let contactPerson = row[col.contactPerson];
  let contactEmail = row[col.contactEmail];

  if (!contactPerson) {

    contactPerson =
      `${row[col.firstName]} ${row[col.lastName]}`.trim();

  }

  if (!contactEmail) {

    contactEmail = row[col.email];

  }

  // ==========================================================
  // NEW — added for Communication Center (Email.gs / AI.gs)
  // ==========================================================
  const ext = getExtendedColumnMap_();

  let reason = row[ext.purpose];
  let erasmusCode = row[ext.erasmusCode];
  // FIX (Bug 1): languageStudents / languageStaff / languageOther
  // come from columns R/S/otherLanguage, which the form fills in
  // regardless of application direction. They must NOT be blanked
  // for STU_TO_FOREIGN applications — same reasoning already
  // applied to erasmusCode below (redirect, don't discard).
  let languageStudents = row[ext.englishStudents];
  let languageStaff = row[ext.englishStaff];
  let languageOther = row[ext.otherLanguage];
  let bip = row[ext.bip];

  if (isSTUApplication) {
    // The U→AB block (fixed indexes) covers the STU→foreign
    // direction of the form. It has its own purpose / erasmus code /
    // "overlap" (BIP) fields, so those are redirected to the
    // foreign-side columns — same pattern as erasmusCode.
    reason = row[col.foreignPurpose];
    erasmusCode = row[col.foreignErasmusCode];
    bip = row[col.foreignOverlap];
    // NOTE: languageStudents/languageStaff/languageOther are
    // intentionally left as-is (from ext.englishStudents /
    // ext.englishStaff / ext.otherLanguage) — those columns are
    // shared across both directions of the form, unlike purpose/
    // erasmusCode/bip which have dedicated STU-side columns.
  }

  const facultyCodes = extractFacultyCodes_(faculty);

  return {

    id: id,

    timestamp: row[col.timestamp],

    email: row[col.email],

    firstName: row[col.firstName],

    lastName: row[col.lastName],

    fullName: `${row[col.firstName]} ${row[col.lastName]}`.trim(),

    status: row[col.status],

    phone: row[col.phone],

    country: row[col.country],

    university: university,

    position: position,

    faculty: faculty,

    mobility: mobility,

    studyLevel: studyLevel,

    contactPerson: contactPerson,

    contactEmail: contactEmail,

    applicationType: isSTUApplication
      ? "STU_TO_FOREIGN"
      : "FOREIGN_TO_STU",

    // ---- NEW fields (additive only) ----
    reason: reason || '',
    erasmusCode: erasmusCode || '',
    languageStudents: languageStudents || '',
    languageStaff: languageStaff || '',
    languageOther: languageOther || '',
    bip: bip || '',
    facultyCodes: facultyCodes

  };

}


function getAllApplications() {

  const rows = getRows();

  return rows.map((row, index) => mapApplication(row, index + 1));

}


function getApplicationById(id) {

  const applications = getAllApplications();

  return applications.find(app => app.id == id) || null;

}

/**
 * Resolves column indexes for the extra fields Email.gs / AI.gs
 * need (applicant's stated reason, Erasmus code, language
 * requirements, BIP flag) WITHOUT touching Database.gs's
 * getColumnMap(). Tries several header spellings, then falls back
 * to a loose substring match so small header edits in the Form
 * don't silently break email generation.
 */
function getExtendedColumnMap_() {

  const headers = getHeaders();

  return {

    purpose: findHeaderIndex_(headers, [
      "Why are you interested in concluding the IIA with foreign university? Please specify the purpose of the IIA (i.e. for which types of mobility it is intended).",
      "Why are you interested in concluding the IIA with foreign university?",
      "Why are you interested in concluding the IIA",
      "Purpose of the IIA"
    ]),

    erasmusCode: findHeaderIndex_(headers, [
      "If applicable, Erasmus code of foreign institution",
      "If applicable, Erasmus code of foreign instituion",
      "Erasmus code of foreign institution",
      "Erasmus code"
    ]),

    // FIX (Bug 2): the most specific candidate ("incoming students" /
    // "incoming employees") must be tried BEFORE the generic
    // "required level of English" phrase. Both headers share that
    // generic phrase, so checking it first made englishStaff match
    // the Students column (whichever header the loose-match loop
    // reaches first). Trying the specific phrase first resolves each
    // field to its own column; the generic phrase remains only as a
    // last-resort fallback.
    englishStudents: findHeaderIndex_(headers, [
        "incoming students",
        "required level of English"
    ]),

    englishStaff: findHeaderIndex_(headers, [
        "incoming employees",
        "required level of English"
    ]),

    otherLanguage: findHeaderIndex_(headers, [
      "If your institution requires knowledge of a language other than English for incoming staff and students, please specify:"
    ]),

    bip: findHeaderIndex_(headers, [
      "Would you like to include BIPs in the agreement?"
    ])

  };

}

/**
 * Tries exact header matches first, then a case/whitespace-loose
 * substring match. Returns -1 (never throws) if nothing matches,
 * so a missing/renamed column degrades to an empty field instead
 * of breaking application loading or email building.
 */
function findHeaderIndex_(headers, candidates) {

  for (let i = 0; i < candidates.length; i++) {
    const idx = headers.indexOf(candidates[i]);
    if (idx !== -1) return idx;
  }

  const normalize = s => String(s).toLowerCase().replace(/\s+/g, ' ').trim();
  const normCandidates = candidates.map(normalize);

  for (let h = 0; h < headers.length; h++) {
    const normHeader = normalize(headers[h]);
    for (let c = 0; c < normCandidates.length; c++) {
      if (normHeader.indexOf(normCandidates[c]) !== -1 ||
          normCandidates[c].indexOf(normHeader) !== -1) {
        return h;
      }
    }
  }

  return -1;

}

/**
 * A single "faculty" answer can contain one or more faculties,
 * e.g. "Faculty of Electrical Engineering and Information
 * Technology (FEI), Faculty of Architecture and Design (FAD)".
 * Extracts just the short codes: ["FEI", "FAD"].
 * Also tolerates answers that are already bare codes ("FEI, FAD").
 * Shared by ApplicationService.gs and Email.gs.
 */
function extractFacultyCodes_(facultyRaw) {

  if (!facultyRaw) return [];

  return String(facultyRaw)
    .split(',')
    .map(part => part.trim())
    .filter(Boolean)
    .map(part => {
      const match = part.match(/\(([^)]+)\)/);
      return (match ? match[1] : part).trim().toUpperCase();
    });

}

/**
 * Auth.gs
 * ------------------------------------------------------------
 * Simple two-role authorization for the STU International Office
 * app: Admin (listed + Active in the AuthorizedUsers sheet) or
 * Viewer (everyone else). Google Sign-In on the frontend proves
 * WHO the person is (their Google ID token); this file is what
 * decides WHAT they're allowed to do, and it is the only place
 * that decision may be trusted — never the frontend.
 *
 * Sheet expected: "AuthorizedUsers"
 *   Column A: Email      Column B: Active (TRUE / FALSE)
 *
 * Add or remove an admin by editing that sheet — no code changes,
 * no redeploy needed.
 * ------------------------------------------------------------
 */

/* Must be the exact same OAuth Web client ID used in auth.js on
   the frontend. Anthropic/Google Cloud Console → APIs & Services
   → Credentials → OAuth 2.0 Client IDs. */
var GOOGLE_CLIENT_ID = "1026351516868-ckgto5ppumua6m2htqcoeuubid3jrnkb.apps.googleusercontent.com";

var AUTH_SHEET_NAME = "AuthorizedUsers";

/**
 * Verifies a Google ID token against Google's tokeninfo endpoint.
 * Confirms it's a real, unexpired token issued for THIS app
 * (matching aud === GOOGLE_CLIENT_ID) and that the email on it is
 * verified by Google. Returns the decoded payload, or null if the
 * token is missing, malformed, expired, or issued for another app.
 */
function verifyGoogleIdToken_(idToken) {
  if (!idToken) return null;
  try {
    var resp = UrlFetchApp.fetch(
      "https://oauth2.googleapis.com/tokeninfo?id_token=" + encodeURIComponent(idToken),
      { muteHttpExceptions: true }
    );
    if (resp.getResponseCode() !== 200) return null;
    var payload = JSON.parse(resp.getContentText());

    if (payload.aud !== GOOGLE_CLIENT_ID) return null;
    if (!payload.email || payload.email_verified !== "true" && payload.email_verified !== true) return null;
    if (payload.exp && Number(payload.exp) < (Date.now() / 1000)) return null;

    return payload;
  } catch (err) {
    console.error("verifyGoogleIdToken_ failed: " + err);
    return null;
  }
}

/**
 * Looks up an email in the AuthorizedUsers sheet. Only rows with
 * Active === TRUE count as Admins; everything else (not present,
 * or present but not Active) is a Viewer.
 */
function isAuthorizedAdmin_(email) {
  if (!email) return false;
  var normalized = String(email).trim().toLowerCase();

  var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(AUTH_SHEET_NAME);
  if (!sheet) {
    console.error("AuthorizedUsers sheet not found.");
    return false;
  }

  var values = sheet.getDataRange().getValues(); // [ [Email, Active], ... ] including header row
  for (var i = 1; i < values.length; i++) {
    var rowEmail = String(values[i][0] || "").trim().toLowerCase();
    if (rowEmail !== normalized) continue;
    var active = values[i][1];
    var isActive = active === true || String(active).trim().toUpperCase() === "TRUE";
    return isActive;
  }
  return false;
}

/**
 * PUBLIC — the endpoint the frontend calls right after Google
 * Sign-In. Wire this into your action router, e.g.:
 *
 *   case 'checkAuthorization':
 *     return successResponse(checkAuthorization(params.idToken));
 *
 * Never returns an error for an unauthorized user — an
 * unauthorized/invalid token simply resolves to the Viewer role,
 * exactly like a signed-out visitor.
 */
function checkAuthorization(idToken) {
  var payload = verifyGoogleIdToken_(idToken);
  if (!payload) {
    return { authorized: false, role: "Viewer" };
  }

  var email = String(payload.email).toLowerCase();
  var admin = isAuthorizedAdmin_(email);

  return {
    authorized: admin,
    role: admin ? "Admin" : "Viewer",
    email: email,
    name: payload.name || "",
    picture: payload.picture || ""
  };
}

/**
 * PUBLIC — call this at the very top of every state-changing
 * endpoint (sendEmail, saveDraft, deleteDraft, ai, ...). Throws if
 * the caller is not an authorized Admin; returns the verified
 * email if they are, in case the handler wants to log who acted.
 *
 * Example integration in an existing handler:
 *
 *   function sendEmail(params) {
 *     try {
 *       requireAdmin(params.idToken);
 *     } catch (err) {
 *       return errorResponse('Access denied.');
 *     }
 *     ... existing sendEmail logic ...
 *   }
 *
 * Do this for sendEmail, saveDraft, deleteDraft, and every AI
 * action (mode:'description', mode:'polish', etc). Never skip
 * this because the frontend already disables the button — the
 * frontend can be bypassed; this cannot.
 */
function requireAdmin(idToken) {
  var auth = checkAuthorization(idToken);
  if (!auth.authorized) {
    throw new Error("Access denied.");
  }
  return auth.email;
}

/**
 * Dashboard Service
 */

function getDashboard() {

  const applications = getAllApplications();

  const dashboard = {

    totalApplications: applications.length,

    newApplications: 0,

    signed: 0,

    rejected: 0,

    inProgress: 0,

    recentApplications: applications
      .slice()
      .sort((a, b) => new Date(b.timestamp) - new Date(a.timestamp))
      .slice(0, 5)

  };

  applications.forEach(app => {

    if (app.status === CONFIG.STATUS.NEW)
      dashboard.newApplications++;

    else if (app.status === CONFIG.STATUS.SIGNED)
      dashboard.signed++;

    else if (app.status === CONFIG.STATUS.REJECTED)
      dashboard.rejected++;

    else if (CONFIG.STATUS.IN_PROGRESS.includes(app.status))
      dashboard.inProgress++;

  });

  return successResponse(dashboard);

}

DraftService.gs

/**
 * DraftService.gs
 * ------------------------------------------------------------
 * FIXED — saveDraft() used to be a stub that always returned a
 * fake "Draft saved." response without writing anything anywhere.
 * It didn't crash (so it wasn't a cause of the CORS symptom), but
 * it also didn't actually save drafts — a refresh or a new session
 * lost everything, which isn't acceptable for a production
 * Communication Center.
 *
 * Since Google Sheets is the only database in this project, drafts
 * are now persisted in a dedicated "Drafts" sheet (auto-created on
 * first use, exactly like nothing else in the project needs to be
 * touched for this). One row per applicationId; saving again for
 * the same application updates that row instead of duplicating it.
 *
 * New functions: getDraftsSheet_ (private helper), getDrafts
 * (list all — used by the "drafts" action so the frontend can
 * restore drafts after a refresh), getDraft (single), deleteDraft
 * (cleanup once an email is actually sent). saveDraft keeps its
 * original name and signature — only its body changed.
 * ------------------------------------------------------------
 */

const DRAFTS_HEADERS_ = ["ApplicationId", "Template", "Recipients", "Subject", "Body", "UpdatedAt"];

/**
 * Returns the Drafts sheet, creating it (with headers) on first use.
 */
function getDraftsSheet_() {

  const ss = SpreadsheetApp.openById(CONFIG.SPREADSHEET_ID);

  let sheet = ss.getSheetByName(CONFIG.SHEETS.DRAFTS);

  if (!sheet) {
    sheet = ss.insertSheet(CONFIG.SHEETS.DRAFTS);
    sheet.appendRow(DRAFTS_HEADERS_);
  }

  return sheet;

}

/**
 * Finds the 1-based row index (including the header row) of the
 * draft for a given applicationId, or -1 if none exists yet.
 */
function findDraftRow_(sheet, applicationId) {

  const values = sheet.getDataRange().getValues();

  for (let i = 1; i < values.length; i++) {
    if (String(values[i][0]) === String(applicationId)) {
      return i + 1; // +1 because sheet rows are 1-based
    }
  }

  return -1;

}

/**
 * POST /saveDraft
 * Body: { applicationId, template, recipients, subject, body }
 * Creates or updates the single draft row for this application.
 */
function saveDraft(data) {

  const applicationId = data && (data.applicationId || data.id);

  if (!applicationId) {
    return errorResponse("Missing applicationId.");
  }

  const sheet = getDraftsSheet_();

  const rowData = [
    applicationId,
    (data.template || "").toString(),
    (data.recipients || "").toString(),
    (data.subject || "").toString(),
    (data.body || "").toString(),
    new Date()
  ];

  const existingRow = findDraftRow_(sheet, applicationId);

  if (existingRow === -1) {
    sheet.appendRow(rowData);
  } else {
    sheet.getRange(existingRow, 1, 1, rowData.length).setValues([rowData]);
  }

  return successResponse({
    message: "Draft saved.",
    applicationId: applicationId
  });

}

/**
 * GET /drafts
 * Returns every saved draft, so the frontend can restore its
 * in-memory state.drafts map after a page refresh.
 */
function getDrafts() {

  const sheet = getDraftsSheet_();

  const values = sheet.getDataRange().getValues();

  values.shift(); // header row

  const drafts = values
    .filter(row => row[0] !== "" && row[0] !== null && row[0] !== undefined)
    .map(row => ({
      applicationId: row[0],
      template: row[1],
      recipients: row[2],
      subject: row[3],
      body: row[4],
      updatedAt: row[5]
    }));

  return successResponse(drafts);

}

/**
 * GET /draft?id=... (or POST with { applicationId })
 * Returns the single saved draft for one application, if any.
 */
function getDraft(applicationId) {

  if (!applicationId) {
    return errorResponse("Missing applicationId.");
  }

  const sheet = getDraftsSheet_();

  const values = sheet.getDataRange().getValues();

  values.shift();

  const row = values.find(r => String(r[0]) === String(applicationId));

  if (!row) {
    return errorResponse("No draft found for this application.");
  }

  return successResponse({
    applicationId: row[0],
    template: row[1],
    recipients: row[2],
    subject: row[3],
    body: row[4],
    updatedAt: row[5]
  });

}

/**
 * POST /deleteDraft
 * Body: { applicationId }
 * Removes the saved draft row, typically called right after an
 * email is successfully sent so the draft doesn't linger.
 * Always succeeds (even if there was nothing to delete) so the
 * frontend's best-effort cleanup call never needs special-casing.
 */
function deleteDraft(applicationId) {

  if (!applicationId) {
    return successResponse({ deleted: false });
  }

  const sheet = getDraftsSheet_();

  const rowIndex = findDraftRow_(sheet, applicationId);

  if (rowIndex === -1) {
    return successResponse({ deleted: false });
  }

  sheet.deleteRow(rowIndex);

  return successResponse({ deleted: true });

}


EmailService.gs

/**
 * Email Service
 * ------------------------------------------------------------
 * FIXED (in addition to Router.gs's global try/catch):
 *
 * 1) buildFacultyEmail() used to do its own manual
 *    .replace(/{{X}}/g, application.x) chain, with NO fallback
 *    for missing values. If e.g. application.position was
 *    undefined, {{POSITION}} was replaced with the literal text
 *    "undefined" in a real e-mail sent to a faculty coordinator.
 *    It now reuses the same replaceVariables() helper the partner
 *    flow already used, which safely falls back to "" for every
 *    field.
 *
 * 2) sendFacultyEmail() and sendPartnerEmail() now validate their
 *    inputs (recipient e-mail format, required subject/body)
 *    instead of handing bad data straight to GmailApp.sendEmail,
 *    which would throw and crash the request. Multiple recipients
 *    still all get the mail (unchanged loop), obviously malformed
 *    addresses are now skipped and reported back instead of
 *    aborting the whole send.
 *
 * Function names/signatures are unchanged, so nothing that calls
 * into this file needs to change.
 * ------------------------------------------------------------
 */

function replaceVariables(template, application) {

  return template

    .replace(/{{FULL_NAME}}/g, application.fullName || "")

    .replace(/{{UNIVERSITY}}/g, application.university || "")

    .replace(/{{COUNTRY}}/g, application.country || "")

    .replace(/{{POSITION}}/g, application.position || "")

    .replace(/{{FACULTY}}/g, application.faculty || "")

    .replace(/{{MOBILITY}}/g, application.mobility || "")

    .replace(/{{STUDY_LEVEL}}/g, application.studyLevel || "")

    .replace(/{{EMAIL}}/g, application.email || "")

    .replace(/{{CONTACT_PERSON}}/g, application.contactPerson || "");

}

/**
 * Very small, deliberately permissive e-mail shape check — good
 * enough to catch empty strings / obvious typos before handing
 * them to GmailApp, without rejecting valid-but-unusual addresses.
 */
function isValidEmail_(email) {
  return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(String(email || "").trim());
}


/**
 * Send email to partner university
 */

function sendPartnerEmail(id) {

  const application = getApplicationById(id);

  if (!application) {
    return errorResponse("Application not found");
  }

  const recipient = (application.contactEmail || application.email || "").toString().trim();

  if (!isValidEmail_(recipient)) {
    return errorResponse("No valid contact e-mail address found for this application.");
  }

  const template = getEmailTemplate("Partner");

  const subject = template.subject;

  const body = replaceVariables(template.body, application);

  GmailApp.sendEmail(
    recipient,
    subject,
    body
  );

  return successResponse({
    message: "Partner email sent.",
    sentTo: recipient
  });

}

/**
 * Send Faculty Email
 * Accepts an array of recipients (preferred) or a comma-separated
 * "to" string (fallback), matching what the Communication Center
 * sends. Sends the same subject/body/htmlBody to every recipient —
 * multiple recipients are fully supported.
 */

function sendFacultyEmail(request) {

  if (!request || !request.applicationId) {
    return errorResponse("Missing applicationId.");
  }

  const application = getApplicationById(request.applicationId);

  if (!application) {
    return errorResponse("Application not found");
  }

  let recipients = request.recipients;

  if (!Array.isArray(recipients) || recipients.length === 0) {
    recipients = String(request.to || "")
      .split(",")
      .map(e => e.trim())
      .filter(Boolean);
  }

  recipients = recipients.map(e => String(e).trim()).filter(Boolean);

  const validRecipients = recipients.filter(isValidEmail_);
  const skippedRecipients = recipients.filter(e => !isValidEmail_(e));

  if (validRecipients.length === 0) {
    return errorResponse("No valid recipient e-mail addresses were provided.");
  }

  const subject = (request.subject || "").toString().trim();

  if (!subject) {
    return errorResponse("Subject is required.");
  }

  const plainBody = request.body || "";
  const htmlBody = request.htmlBody || "";

  if (!plainBody && !htmlBody) {
    return errorResponse("Email body is required.");
  }

  const options = {};
  if (htmlBody) {
    options.htmlBody = htmlBody;
  }

  validRecipients.forEach(email => {

    GmailApp.sendEmail(
      email,
      subject,
      plainBody,
      options
    );

  });

  return successResponse({

    sentTo: validRecipients,

    skipped: skippedRecipients,

    totalRecipients: validRecipients.length,

    message: "Emails sent successfully."

  });

}

/**
 * Builds email from template
 */

function buildFacultyEmail(id) {

  const application = getApplicationById(id);

  if (!application) {

    return errorResponse("Application not found");

  }

  const template = getEmailTemplate("Faculty");

  const subject = template.subject;

  const body = replaceVariables(template.body, application);

  return successResponse({

    subject,

    body,

    recipients: getFacultyEmails(application.faculty)

  });

}

function buildPartnerEmail(id) {

  const application = getApplicationById(id);

  if (!application) {
    return errorResponse("Application not found");
  }

  const template = getEmailTemplate("Partner");

  const subject = template.subject;

  const body = replaceVariables(template.body, application);

  return successResponse({
      subject,
      body
  });

}

/**
 * Faculty Service
 */

function getFacultyEmails(facultyString) {

  if (!facultyString) return [];

  const sheet = SpreadsheetApp
    .openById(CONFIG.SPREADSHEET_ID)
    .getSheetByName(CONFIG.SHEETS.FACULTIES);

  const data = sheet.getDataRange().getValues();

  data.shift();

  const selectedFaculties = facultyString
    .split(/\s*,\s*/)
    .map(f => f.trim());

  const emails = [];

  data.forEach(row => {

    const faculty = String(row[0]).trim();

    const email = row[2];

    if (selectedFaculties.includes(faculty)) {

      emails.push(email);

    }

  });

  return [...new Set(emails)];

}

function getFaculties() {

  const sheet = SpreadsheetApp
    .openById(CONFIG.SPREADSHEET_ID)
    .getSheetByName(CONFIG.SHEETS.FACULTIES);

  const rows = sheet.getDataRange().getValues();

  rows.shift();

  const result = rows.map(r => ({
    faculty: r[0],
    coordinator: r[1],
    email: r[2]
  }));

  return successResponse(result);

}

Router.gs

/**
 * Router.gs
 * ------------------------------------------------------------
 * FIXED — root cause of "doPost FAILED" / browser CORS errors:
 *
 * Google Apps Script Web Apps only send back the automatic
 * Access-Control-Allow-Origin header when the function running
 * behind doGet/doPost returns NORMALLY (i.e. via ContentService).
 * If ANY code on the way throws an uncaught exception — a missing
 * function, a bad request, an external API error, a malformed
 * JSON body, etc. — Apps Script serves its own error page instead,
 * and that page has NO CORS header at all. The browser then
 * reports "blocked by CORS policy: No Access-Control-Allow-Origin
 * header", even though the real problem is a server-side crash.
 * That is exactly what the Apps Script "Executions" panel shows
 * as doPost → FAILED.
 *
 * The fix is structural: routeRequest() now wraps everything in
 * try/catch (including the JSON.parse of the POST body) so that
 * no matter what goes wrong inside a handler, the function always
 * returns a normal errorResponse() — which always carries the
 * CORS header — instead of throwing.
 *
 * CHANGED — mail sending moved out of Apps Script entirely:
 * emails are now sent directly from the browser to the Gmail
 * REST API using each Admin's own OAuth access token (see
 * auth.js: requestGmailAccessToken(), and communication.html:
 * buildMimeMessage()/sendGmailMessage()). Apps Script keeps only
 * Sheets, AI, drafts, statistics and authorization, per the
 * GmailApp-based sendEmail / sendPartnerEmail routes below are
 * intentionally removed — GmailApp used to send as the script
 * owner's account, which is exactly what this change replaces.
 * If something still posts to these actions (e.g. a stale cached
 * page), it gets a clear error instead of silently sending mail
 * from the wrong account.
 * ------------------------------------------------------------
 */

function routeRequest(e) {

  try {

    let action = e.parameter.action || "";

    let request = {};

    if (e.postData && e.postData.contents) {

      try {
        request = JSON.parse(e.postData.contents);
      } catch (parseErr) {
        return errorResponse("Invalid JSON in request body: " + parseErr.message);
      }

      action = request.action || action;

    }

    switch (action) {

      case "checkAuthorization":
        return successResponse(
          checkAuthorization(request.idToken)
        );

      case "applications":
        return getApplications();

      case "application":
        return getApplication(e.parameter.id);

      case "dashboard":
        return getDashboard();

      case "statistics":
        return getStatistics();

      case "map":
        return getMapData();

      case "faculties":
        return getFaculties();

      case "emailTemplates":
        return getEmailTemplates();

      case "sender":
        return getSender();

      case "saveDraft":
        return saveDraft(request);

      case "drafts":
        return getDrafts();

      case "draft":
        return getDraft(e.parameter.id || request.applicationId);

      case "deleteDraft":
        return deleteDraft(request.applicationId || e.parameter.id);

      case "search":
        return searchApplications(e.parameter.q);

      case "buildEmail":
        return buildFacultyEmail(e.parameter.id);

      case "buildPartnerEmail":
        return buildPartnerEmail(e.parameter.id);

      /* REMOVED: "sendEmail" and "sendPartnerEmail" used to call
         GmailApp.sendEmail(), which always sends as the Apps
         Script deployment owner, not the signed-in Admin. Sending
         now happens client-side against the Gmail REST API
         instead (see communication.html). Kept here only to give
         a clear, actionable error if anything still calls them. */
      case "sendEmail":
      case "sendPartnerEmail":
        return errorResponse(
          "This app no longer sends email through Apps Script. Emails are sent " +
          "directly from the signed-in Admin's own Google account via the Gmail " +
          "API — please reload the page."
        );

      case "ai": {

        switch (request.mode) {

          case "description":
            return successResponse(
              generateAISummary(request.text)
            );

          case "polish":
            return successResponse(
              improveEmail(request.text)
            );

          default:
            return errorResponse(
              "Unknown AI mode: " + request.mode
            );

        }

      }

      default:
        return errorResponse("Unknown API endpoint");

    }

  } catch (err) {

    // Last line of defense: whatever broke above, always answer with a
    // normal JSON response (so CORS headers are always present) instead
    // of letting the exception reach the Apps Script runtime, which is
    // what produced the "doPost FAILED" / CORS-looking browser errors.
    return errorResponse(err && err.message ? err.message : String(err));

  }

}

/**
 * GET /statistics

function getStatistics() {

  const applications = getAllApplications();

  const statistics = {

    totalApplications: applications.length,

    byStatus: {},

    byCountry: {},

    byFaculty: {},

    byMobility: {},

    byStudyLevel: {}

  };

  applications.forEach(app => {

    increment(statistics.byStatus, app.status);

    increment(statistics.byCountry, app.country);

    splitAndCount(statistics.byFaculty, app.faculty);

    splitAndCount(statistics.byMobility, app.mobility);

    splitAndCount(statistics.byStudyLevel, app.studyLevel);

  });

  return successResponse(statistics);

}


TemplateService.gs

/**
 * Template Service
 * ------------------------------------------------------------
 * FIXED — Router.gs's "emailTemplates" action called
 * getEmailTemplates() (plural), but only getEmailTemplate(name)
 * (singular) existed anywhere in the project. Calling the
 * "emailTemplates" endpoint therefore always threw
 * "ReferenceError: getEmailTemplates is not defined", crashing
 * the request. Added below, additive only — getEmailTemplate()
 * is untouched and still used exactly as before by
 * buildFacultyEmail / buildPartnerEmail / sendPartnerEmail.
 * ------------------------------------------------------------
 */

/**
 * Returns email template by name
 */

function getEmailTemplate(templateName) {

  const sheet = SpreadsheetApp
    .openById(CONFIG.SPREADSHEET_ID)
    .getSheetByName(CONFIG.SHEETS.EMAIL_TEMPLATE);

  const rows = sheet.getDataRange().getValues();

  rows.shift(); // Remove header

  const row = rows.find(r => r[0] === templateName);

  if (!row) {
    throw new Error("Template not found: " + templateName);
  }

  return {
    subject: row[1],
    body: row[2]
  };

}

/**
 * GET /emailTemplates
 * Returns every row of the EmailTemplate sheet as a flat list,
 * e.g. [{ template:"Faculty", subject:"...", body:"..." },
 *       { template:"Partner", subject:"...", body:"..." }].
 * Used by the Communication Center to (optionally) override its
 * local fallback templates with the live sheet content.
 */
function getEmailTemplates() {

  const sheet = SpreadsheetApp
    .openById(CONFIG.SPREADSHEET_ID)
    .getSheetByName(CONFIG.SHEETS.EMAIL_TEMPLATE);

  const rows = sheet.getDataRange().getValues();

  rows.shift(); // Remove header

  const templates = rows
    .filter(r => r[0])
    .map(r => ({
      template: r[0],
      subject: r[1],
      body: r[2]
    }));

  return successResponse(templates);

}


Utils.gs

/**
 * Utility Functions
 */

function increment(object, key) {

  if (!key) return;

  key = String(key).trim();

  object[key] = (object[key] || 0) + 1;

}

function splitAndCount(object, value) {

  if (!value) return;

  String(value)

    .split(",")

    .map(item => item.trim())

    .filter(Boolean)

    .forEach(item => increment(object, item));

}
