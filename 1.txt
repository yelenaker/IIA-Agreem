/**
 * JSON Responses
 */

function jsonResponse(data) {

  return ContentService
    .createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);

}

function successResponse(data) {

  return jsonResponse({

    success: true,

    timestamp: new Date(),

    data

  });

}

function errorResponse(message) {

  return jsonResponse({

    success: false,

    timestamp: new Date(),

    error: message

  });

}
